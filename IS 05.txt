from math import gcd, fmod

p = 3
q = 7

n = p * q
print("n =", n)

phi = (p - 1) * (q - 1)

e = 2
while e < phi:
    if gcd(e, phi) == 1:
        break
    e += 1
print("e =", e)

k = 2
d = ((k * phi) + 1) // e
print("d =", d)
print("Public key: (%d, %d)" % (e, n))
print("Private key: (%d, %d)" % (d, n))

msg = 11
print("Original message:", msg)

C = pow(msg, e, n)
print("Encrypted message:", C)

M = pow(C, d, n)
print("Decrypted message:", M)
